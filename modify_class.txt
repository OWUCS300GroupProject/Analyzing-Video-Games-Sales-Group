//used to change numerical data into the classifications we want
//used on the training data
void Modify_Class(vector<Instance>& examples){

	for(int i = 0; i < examples.size(); i++){
		
		double cur_val = examples[i].Get_Numeric_Value(examples[i].Get_Num_Attributes()-1);
		double replace = floor(cur_val/5);
		
		//new category is the highest multiple of 5 that can go into the previous sales price
		examples[i].Add_Value(examples[i].Get_Num_Attributes()-1, to_string(replace));
		
	}
	
}
